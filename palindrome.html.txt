      <script>
        function palin(){
        let input=document.querySelector("#input").value.toLowerCase().replace(/\s+/g,"")
        let ans
        let rev=input.split("").reverse().join("")
        if(input==rev){
          ans="palindrome"
        }
        else{
          ans="Not a palindrome"
        }
        let result=document.getElementById("result")
        result.innerHTML="It is "+ans
        }
      </script>
